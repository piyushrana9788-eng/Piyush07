Q7:Write a C++ program to convert a decimal number to binary numbers

int main() {
    int num, binary[32], i = 0;

    cout << "Enter decimal number: ";
    cin >> num;

    while(num > 0) {
        binary[i] = num % 2;
        num = num / 2;
        i++;
    }

    cout << "Binary = ";

    }

    return 0;

Q20:Write a C++ program to compute the sum of the first 100 prime numbers 


#include<iostream>
using namespace std;

bool isPrime(int num){
    if(num < 2) return false;
    for(int i=2;i*i<=num;i++)
        if(num%i==0) return false;
    return true;
}

int main() {
    int count=0, num=2;
    long sum=0;

    while(count< 100){
        if(isPrime(num)){
            sum += num;
            count++;
        }
        num++;
    }

    cout << "Sum of first 100 prime numbers: " << sum;
    return 0;
}  

OUTPUT:
Sum of first 100 prime numbers: 24133

Q113:Create a class entering the command line arguments from the user and show all the arguments as output.


#include<iostream>
using namespace std;

class Demo {
public:
    void display(int argc, char* argv[]) {
        cout << "Arguments are:\n";
        for(int i=0; i< argc; i++) {
            cout << argv[i] << endl;
        }
    }
};

int main(int argc, char* argv[]) {
    Demo obj;
    obj.display(argc, argv);
}

OUTPUT:
Arguments are:
