Q1: Define a class Complex to represent complex numbers. Overload the + operator to add two complex numbers.


#include<iostream>
using namespace std;

class Complex {
    int real, imag;

public:
    Complex(int r = 0, int i = 0) {
        real = r;
        imag = i;
    }

    Complex operator+(Complex c) {
        return Complex(real + c.real, imag + c.imag);
    }

    void display() {
        cout << real << " + " << imag << "i" << endl;
    }
};

int main() {
    Complex c1(2, 3), c2(4, 5);
    Complex c3 = c1 + c2;
    c3.display();
}

OUTPUT:
6 + 8i

Q2:Create a class Fraction to represent fractions. Overload the + operator to add two fractions.


#include<iostream>
using namespace std;

class Fraction {
    int num, den;

public:
    Fraction(int n = 0, int d = 1) {
        num = n;
        den = d;
    }

    Fraction operator+(Fraction f) {
        int n = num * f.den + f.num * den;
        int d = den * f.den;
        return Fraction(n, d);
    }

    void display() {
        cout << num << "/" << den << endl;
    }
};

int main() {
    Fraction f1(1, 2), f2(3, 4);
    Fraction f3 = f1 + f2;
    f3.display();
}

Q3:Define a class Matrix to represent matrices. Overload the * operator to multiply two matrices.


#include<iostream>
using namespace std;

class Matrix {
    int a[2][2];

public:
    void input() {
        for(int i=0;i< 2;i++)
            for(int j=0;j< 2;j++)
                cin >> a[i][j];
    }

    Matrix operator*(Matrix m) {
        Matrix result;
        for(int i=0;i< 2;i++) {
            for(int j=0;j< 2;j++) {
                result.a[i][j] = 0;
                for(int k=0;k< 2;k++) {
                    result.a[i][j] += a[i][k] * m.a[k][j];
                }
            }
        }
        return result;
    }

    void display() {
        for(int i=0;i< 2;i++) {
            for(int j=0;j< 2;j++)
                cout << a[i][j] << " ";
            cout << endl;
        }
    }
};

int main() {
    Matrix m1, m2, m3;
    cout << "Enter first matrix:\n";
    m1.input();
    cout << "Enter second matrix:\n";
    m2.input();

    m3 = m1 * m2;
    cout << "Result:\n";
    m3.display();
}

Result:

Q5:Define a class Date to represent dates. Overload the < operator to compare two dates.



#include 
using namespace std;

class Date {
    int d, m, y;

public:
    Date(int day, int month, int year) {
        d = day; m = month; y = year;
    }

    bool operator<(Date dt) {
        if(y < dt.y) return true;
        if(y == dt.y && m < dt.m) return true;
        if(y == dt.y && m == dt.m && d < dt.d) return true;
        return false;
    }
};

int main() {
    Date d1(10, 5, 2023), d2(12, 6, 2023);

    if(d1 < d2)
        cout << "d1 is earlier\n";
    else
        cout << "d2 is earlier\n";
}

OUTPUT:
d1 is earlier

Q6:Implement a class Money to represent money. Overload the + operator to add two amounts of money.


#include<iostream>
using namespace std;

class Money {
    int rupees, paise;

public:
    Money(int r = 0, int p = 0) {
        rupees = r;
        paise = p;
    }

    Money operator+(Money m) {
        Money temp;
        temp.paise = paise + m.paise;
        temp.rupees = rupees + m.rupees + temp.paise / 100;
        temp.paise %= 100;
        return temp;
    }

    void display() {
        cout << "Rs " << rupees << "." << paise << endl;
    }
};

int main() {
    Money m1(10, 75), m2(5, 50);
    Money m3 = m1 + m2;
    m3.display();
}
OUTPUT:
Rs 16.25

Q7:Create a class Rectangle to represent rectangles. Overload the == operator to check if two rectangles are equal.


#include<iostream>
using namespace std;

class Rectangle {
    int length, width;

public:
    Rectangle(int l, int w) {
        length = l;
        width = w;
    }

    bool operator==(Rectangle r) {
        return (length == r.length && width == r.width);
    }
};

int main() {
    Rectangle r1(10, 5), r2(10, 5);

    if(r1 == r2)
        cout << "Rectangles are equal\n";
    else
        cout << "Rectangles are not equal\n";
}

OUTPUT:
Rectangles are equal

Q8:Define a class Time to represent time. Overload the << operator to display the time in HH:MM format.


#include<iostream>
using namespace std;

class Time {
    int h, m;

public:
    Time(int hour, int min) {
        h = hour;
        m = min;
    }

    friend ostream& operator<<(ostream &out, Time t) {
        out << (t.h < 10 ? "0" : "") << t.h << ":"
            << (t.m < 10 ? "0" : "") << t.m;
        return out;
    }
};

int main() {
    Time t1(9, 5);
    cout << t1;
}

OUTPUT:
09:05
