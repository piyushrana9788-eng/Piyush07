Q2:Program to find the frequency of each element in the array


#include<iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    int a[n];
    bool visited[n];

    cout << "Enter elements: ";
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        visited[i] = false;
    }

    for(int i = 0; i < n; i++) {
        if(visited[i] == false) {
            int count = 1;
            for(int j = i + 1; j < n; j++) {
                if(a[i] == a[j]) {
                    count++;
                    visited[j] = true;
                }
            }
            cout << a[i] << " appears " << count << " time(s)" << endl;
        }
    }

    return 0;
}

OUTPUT:
Enter number of elements: 2
Enter elements: 23
12
23 appears 1 time(s)
12 appears 1 time(s)

Q3:Program to left rotate the elements of an array


#include<iostream>
using namespace std;

int main() {
    int n;

    cin >> n;

    int a[n];
    cout << "Enter elements: ";
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int times;
    cout << "Enter number of times to left rotate: ";
    cin >> times;

    for(int t = 0; t < times; t++) {
        int first = a[0];
        for(int i = 0; i < n - 1; i++) {
            a[i] = a[i + 1];
        }
        a[n - 1] = first;
    }

    cout << "After left rotation: ";
    for(int i = 0; i < n; i++) {
        cout << a[i] << " ";
    }
    cout << endl;

    return 0;
}

Q14:Program to sort the elements of an array in ascending order


#include<iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    int a[n];
    cout << "Enter elements: ";
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }

    // Bubble sort ascending
    for(int i = 0; i < n - 1; i++) {
        for(int j = 0; j < n - i - 1; j++) {
            if(a[j] > a[j + 1]) {
                int temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }

    cout << "Sorted array (ascending): ";
    for(int i = 0; i < n; i++) {
        cout << a[i] << " ";
    }
    cout << endl;

    return 0;
}


OUTPUT:
Enter number of elements: 2
Enter elements: 12
34
Sorted array (ascending): 12 34

Q31:Write a C++ program to reverse an array of integer values


#include<iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    int a[n];
    cout << "Enter elements: ";
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }

    
    int start = 0, end = n - 1;
    while(start < end) {
        int temp = a[start];
        a[start] = a[end];
        a[end] = temp;
        start++;
        end--;
    }

    cout << "Reversed array: ";
    for(int i = 0; i < n; i++) {
        cout << a[i] << " ";
    }
    cout << endl;

    return 0;
}
