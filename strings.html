#include<iostream>
#include 

using namespace std;


bool isAnagram(string first, string second){

    if (first.length() != second.length()){
        return false;
    }

    int freq[26] = {0};

    for (int i = 0; i < first.length(); i++){
        freq[first[i] - 'a']++;
        freq[second[i] - 'a']--;
    }

    for (int i = 0; i < 26; i++){
        if (freq[i] != 0){
            return false;
        }
    }

    return true;
}


int main(){

    string first;
    string second;

    cout << "Enter first string: ";
    cin >> first;

    cout << "Enter second string: ";
    cin >> second;

    if (isAnagram(first, second)){
        cout << "Anagram";
    }

    if (!isAnagram(first, second)){
        cout << "Not anagram";
    }

    return 0;
}

Q12:Program to count the total number of vowels and consonants in a string

bool isLetter(char ch){

    if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')){
        return true;
    }

    return false;
}


bool isVowel(char ch){

    if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' || ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U'){
        return true;
    }

    return false;
}


int main(){

    string text;
    int vowels = 0;
    int consonants = 0;

    cout << "Enter a string: ";
    getline(cin, text);

    for (int i = 0; i < (int)text.length(); i++){
        if (isLetter(text[i])){
            if (isVowel(text[i])){
                vowels++;
            }

            if (!isVowel(text[i])){
                consonants++;
            }
        }
    }

    cout << "Vowels: " << vowels << "\n";
    cout << "Consonants: " << consonants << "\n";
    return 0;
}

Q13:Program to determine whether a given string is palindrome


#include 
#include 

using namespace std;


bool isPalindrome(string text){

    int left = 0;
    int right = (int)text.length() - 1;

    while (left < right){
        if (text[left] != text[right]){
            return false;
        }

        left++;
        right--;
    }

    return true;
}

        cout << "Palindrome\n";

Q14:Program to determine whether one string is a rotation of another


#include 
#include 

using namespace std;


bool containsText(string text, string pattern){

    if (pattern.length() > text.length()){
        return false;
    }

    for (int i = 0; i <= (int)text.length() - (int)pattern.length(); i++){
        int match = 1;

        for (int j = 0; j < (int)pattern.length(); j++){
            if (text[i + j] != pattern[j]){
                match = 0;
                break;
            }
        }

        if (match == 1){
            return true;
        }
    }

    return false;
}


int main(){

    string first;
    string second;

    cout << "Enter first string: ";
    cin >> first;

    cout << "Enter second string: ";
    cin >> second;

    if (first.length() != second.length()){
        cout << "Not rotation\n";
        return 0;
    }

    string doubled = first + first;

    if (containsText(doubled, second)){
        cout << "Rotation\n";
    }

    if (!containsText(doubled, second)){
        cout << "Not rotation\n";
    }

    return 0;
}

Q17:Program to find all the permutations of a string


#include 
#include 

using namespace std;


void swapChar(char &a, char &b){

    char temp = a;
    a = b;
    b = temp;
}


void printPermutations(string &text, int start, int end){

    if (start == end){
        cout << text << "\n";
        return;
    }

    for (int i = start; i <= end; i++){
        swapChar(text[start], text[i]);
        printPermutations(text, start + 1, end);
        swapChar(text[start], text[i]);
    }
}


int main(){

    string text;

    cout << "Enter a string: ";
    cin >> text;

    cout << "Permutations are:\n";
    printPermutations(text, 0, (int)text.length() - 1);

    return 0;
}

Q18:Program to find maximum and minimum occurring character in a string


#include 
#include 

using namespace std;


int main(){

    string text;
    int freq[256] = {0};

    cout << "Enter a string: ";
    getline(cin, text);

    if (text.length() == 0){
        cout << "Empty string\n";
        return 0;
    }

    for (int i = 0; i < (int)text.length(); i++){
        if (text[i] != ' '){
            freq[(int)text[i]]++;
        }
    }

    int maxCount = 0;
    int minCount = 0;
    char maxChar = ' ';
    char minChar = ' ';

    for (int i = 0; i < 256; i++){
        if (freq[i] > 0){
            maxCount = freq[i];
            minCount = freq[i];
            maxChar = (char)i;
            minChar = (char)i;
            break;
        }
    }

    for (int i = 0; i < 256; i++){
        if (freq[i] > 0){
            if (freq[i] > maxCount){
                maxCount = freq[i];
                maxChar = (char)i;
            }

            if (freq[i] < minCount){
                minCount = freq[i];
                minChar = (char)i;
            }
        }
    }

    cout << "Maximum occurring character: " << maxChar << "\n";
    cout << "Minimum occurring character: " << minChar << "\n";
    return 0;
}

Maximum occurring character: h
Minimum occurring character: h

Q21:Program to find the duplicate words in a string


#include 
#include 

using namespace std;


int main(){

    string text;
    string words[200];
    int wordCount = 0;

    cout << "Enter a string: ";
    getline(cin, text);

    string current = "";
    for (int i = 0; i <= (int)text.length(); i++){
        if (i < (int)text.length() && text[i] != ' '){
            current += text[i];
        }

        if (i == (int)text.length() || text[i] == ' '){
            if (current.length() > 0){
                words[wordCount] = current;
                wordCount++;
                current = "";
            }
        }
    }

    cout << "Duplicate words are:\n";
    int found = 0;

    for (int i = 0; i < wordCount; i++){
        int alreadyPrinted = 0;
        for (int k = 0; k < i; k++){
            if (words[k] == words[i]){
                alreadyPrinted = 1;
                break;
            }
        }

        if (alreadyPrinted == 1){
            continue;
        }

        int count = 0;
        for (int j = 0; j < wordCount; j++){
            if (words[i] == words[j]){
                count++;
            }
        }

        if (count > 1){
            cout << words[i] << " ";
            found = 1;
        }
    }

    if (found == 0){
        cout << "No duplicate words";
    }

    cout << "\n";
    return 0;
}

Q24:Program to find the longest repeating sequence in a string


#include 
#include 

using namespace std;


int main(){

    string text;
    string longest = "";

    cout << "Enter a string: ";
    cin >> text;

    int n = (int)text.length();

    for (int len = 1; len <= n / 2; len++){
        for (int i = 0; i + 2 * len <= n; i++){
            int same = 1;

            for (int j = 0; j < len; j++){
                if (text[i + j] != text[i + len + j]){
                    same = 0;
                    break;
                }
            }

            if (same == 1 && len > (int)longest.length()){
                longest = text.substr(i, len);
            }
        }
    }

    if (longest.length() == 0){
        cout << "No repeating sequence found\n";
    }

    if (longest.length() > 0){
        cout << "Longest repeating sequence: " << longest << "\n";
    }

    return 0;
}

        for (int j = 0; j < wordCount; j++){

                count++;
            }
        }

        if (count > maxCount){
            maxCount = count;
            maxWord = words[i];
        }
    }

    cout << "Most repeated word: " << maxWord << "\n";
    cout << "Frequency: " << maxCount << "\n";
    return 0;
}

bool isPalindromeWord(string word){

    int left = 0;
    int right = (int)word.length() - 1;

    while (left < right){
        if (word[left] != word[right]){
            return false;
        }

        left++;
        right--;
    }

    return true;
}


int main(){

    string text;
    string smallest = "";
    string biggest = "";

    cout << "Enter a string: ";
    getline(cin, text);

    string current = "";
    for (int i = 0; i <= (int)text.length(); i++){
        if (i < (int)text.length() && text[i] != ' '){
            current += text[i];
        }

        if (i == (int)text.length() || text[i] == ' '){
            if (current.length() > 0){
                if (isPalindromeWord(current)){
                    if (smallest.length() == 0 || (int)current.length() < (int)smallest.length()){
                        smallest = current;
                    }

                    if ((int)current.length() > (int)biggest.length()){
                        biggest = current;
                    }
                }

                current = "";
            }
        }
    }

    if (smallest.length() == 0){
        cout << "No palindrome word found\n";
        return 0;
    }

    cout << "Smallest palindrome word: " << smallest << "\n";
