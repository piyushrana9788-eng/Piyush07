Q10:Write a C++ program to split a large text file into smaller files of equal size


#include<iostream>
#include 
using namespace std;

int main() {

    ifstream data("data.txt");
    ofstream file1("split1.txt");
    ofstream file2("split2.txt");
    ofstream file3("split3.txt");

    if (!data || !file1 || !file2 || !file3) {
        cout << "Files not opened";
        return 0;
    }

    cout << "Files opened\n";

    data.seekg(0, ios::end);
    int size = data.tellg();
    data.seekg(0, ios::beg);

    int f = size / 3;

    char a;
    int count = 0;

    while (data.get(a)) {
        if (count < f)
            file1 << a;
        else if (count < 2 * f)
            file2 << a;
        else
            file3 << a;

        count++;
    }

    cout << "File Splitted";
}



OUTPUT:
Files opened
File Split Successfully

Q15:Write a C++ program to calculate the average of numbers stored in a file.


#include 
#include 
using namespace std;

int main() {
    ofstream fout("numbers.txt");
    int num;

    // Writing numbers into file
    cout << "Enter 5 numbers:\n";
    for(int i = 0; i < 5; i++) {
        cin >> num;
        fout << num << " ";
    }
    fout.close();

    // Reading and calculating average
    ifstream fin("numbers.txt");
    int sum = 0, count = 0;

    while(fin >> num) {
        sum += num;
        count++;
    }
    fin.close();

    float avg = (float)sum / count;
    cout << "Average = " << avg;

    return 0;
}

Q16:Write a C++ program to store the information of 10 Students having rollno,name,marks,mobileno properties, get() and set() member functions as binary file with following options 1. Insert Record 2. Delete Record 3. Update Record (based on rollno) 4. Search Record (based on rollno) 5. Clone the file 6. Exit.


#include 
#include 
using namespace std;

class Student {
    int rollno;
    char name[50];
    float marks;
    char mobileno[15];

public:
    void get() {
        cout << "Enter Roll No: ";
        cin >> rollno;
        cout << "Enter Name: ";
        cin >> name;
        cout << "Enter Marks: ";
        cin >> marks;
        cout << "Enter Mobile No: ";
        cin >> mobileno;
    }

    void show() {
        cout << "\nRoll No: " << rollno
             << "\nName: " << name
             << "\nMarks: " << marks
             << "\nMobile: " << mobileno << endl;
    }

    int getRoll() {
        return rollno;
    }

    void setMarks(float m) {
        marks = m;
    }
};

void insertRecord() {
    ofstream fout("student.dat", ios::binary | ios::app);
    Student s;
    s.get();
    fout.write((char*)&s, sizeof(s));
    fout.close();
}

void displayAll() {
    ifstream fin("student.dat", ios::binary);
    Student s;
    while(fin.read((char*)&s, sizeof(s))) {
        s.show();
    }
    fin.close();
}

void searchRecord(int r) {
    ifstream fin("student.dat", ios::binary);
    Student s;
    while(fin.read((char*)&s, sizeof(s))) {
        if(s.getRoll() == r) {
            cout << "Record Found:";
            s.show();
            return;
        }
    }
    cout << "Record Not Found\n";
    fin.close();
}

void deleteRecord(int r) {
    ifstream fin("student.dat", ios::binary);
    ofstream fout("temp.dat", ios::binary);
    Student s;

    while(fin.read((char*)&s, sizeof(s))) {
        if(s.getRoll() != r)
            fout.write((char*)&s, sizeof(s));
    }

    fin.close();
    fout.close();

    remove("student.dat");
    rename("temp.dat", "student.dat");
}

void updateRecord(int r) {
    fstream file("student.dat", ios::binary | ios::in | ios::out);
    Student s;

    while(file.read((char*)&s, sizeof(s))) {
        if(s.getRoll() == r) {
            float newMarks;
            cout << "Enter new marks: ";
            cin >> newMarks;
            s.setMarks(newMarks);

            int pos = -1 * sizeof(s);
            file.seekp(pos, ios::cur);
            file.write((char*)&s, sizeof(s));

            cout << "Record Updated\n";
            return;
        }
    }
    cout << "Record Not Found\n";
    file.close();
}

void cloneFile() {
    ifstream fin("student.dat", ios::binary);
    ofstream fout("clone.dat", ios::binary);
    Student s;

    while(fin.read((char*)&s, sizeof(s))) {
        fout.write((char*)&s, sizeof(s));
    }

    cout << "File Cloned Successfully\n";

    fin.close();
    fout.close();
}

int main() {
    int choice, roll;

    do {
        cout << "\n1.Insert\n2.Delete\n3.Update\n4.Search\n5.Clone\n6.Display\n7.Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch(choice) {
            case 1: insertRecord(); break;
            case 2:
                cout << "Enter Roll No: ";
                cin >> roll;
                deleteRecord(roll);
                break;
            case 3:
                cout << "Enter Roll No: ";
                cin >> roll;
                updateRecord(roll);
                break;
            case 4:
                cout << "Enter Roll No: ";
                cin >> roll;
                searchRecord(roll);
                break;
            case 5: cloneFile(); break;
            case 6: displayAll(); break;
        }

    } while(choice != 7);

    return 0;
}
