Q9:Write a C++ program to create a vehicle class hierarchy. The base class should be Vehicle, with derived classes Truck, Car and Motorcycle. Each derived class should have properties such as make, model, year, and fuel type. Implement methods for calculating fuel efficiency, distance traveled, and maximum speed.


#include<iostream>
using namespace std;


class Vehicle {
protected:
    string make, model, fuelType;
    int year;

public:
    void setDetails(string mk, string mdl, int yr, string fuel) {
        make = mk;
        model = mdl;
        year = yr;
        fuelType = fuel;
    }

    virtual void fuelEfficiency() {
        cout << "Fuel efficiency calculation not defined.\n";
    }

    virtual void distanceTraveled(float fuelUsed) {
        cout << "Distance calculation not defined.\n";
    }

    virtual void maxSpeed() {
        cout << "Max speed not defined.\n";
    }
};


class Car : public Vehicle {
public:
    void fuelEfficiency() {
        cout << "Car Fuel Efficiency: 15 km/l\n";
    }

    void distanceTraveled(float fuelUsed) {
        cout << "Car Distance: " << fuelUsed * 15 << " km\n";
    }

    void maxSpeed() {
        cout << "Car Max Speed: 180 km/h\n";
    }
};


class Truck : public Vehicle {
public:
    void fuelEfficiency() {
        cout << "Truck Fuel Efficiency: 8 km/l\n";
    }

    void distanceTraveled(float fuelUsed) {
        cout << "Truck Distance: " << fuelUsed * 8 << " km\n";
    }

    void maxSpeed() {
        cout << "Truck Max Speed: 120 km/h\n";
    }
};


class Motorcycle : public Vehicle {
public:
    void fuelEfficiency() {
        cout << "Motorcycle Fuel Efficiency: 40 km/l\n";
    }

    void distanceTraveled(float fuelUsed) {
        cout << "Motorcycle Distance: " << fuelUsed * 40 << " km\n";
    }

    void maxSpeed() {
        cout << "Motorcycle Max Speed: 100 km/h\n";
    }
};


int main() {
    Car car;
    Truck truck;
    Motorcycle bike;

    car.setDetails("Honda", "City", 2022, "Petrol");
    truck.setDetails("Tata", "LPT", 2020, "Diesel");
    bike.setDetails("Yamaha", "R15", 2023, "Petrol");

    cout << "\n--- Car Details ---\n";
    car.fuelEfficiency();
    car.distanceTraveled(10);
    car.maxSpeed();

    cout << "\n--- Truck Details ---\n";
    truck.fuelEfficiency();
    truck.distanceTraveled(10);
    truck.maxSpeed();

    cout << "\n--- Motorcycle Details ---\n";
    bike.fuelEfficiency();
    bike.distanceTraveled(10);
    bike.maxSpeed();

    return 0;
}

OUTPUT:

Car Max Speed: 180 km/h

Truck Distance: 80 km

Q10:Write a C++ program that creates a class hierarchy for employees of a company. The base class should be Employee, with derived classes Manager, Developer, and Programmer. Each derived class should have properties such as name, address, salary, and job title. Implement methods for calculating bonuses, generating performance reports, and managing projects.


#include<iostream>
using namespace std;

class Employee {
protected:
    string name, address, jobTitle;
    double salary;

public:
    void setDetails(string n, string addr, double sal, string job) {
        name = n;
        address = addr;
        salary = sal;
        jobTitle = job;
    }

    virtual void calculateBonus() {
        cout << "Bonus calculation not defined.\n";
    }

    virtual void performanceReport() {
        cout << "Performance report not available.\n";
    }

    virtual void manageProject() {
        cout << "Project management not defined.\n";
    }
};

class Manager : public Employee {
public:
    void calculateBonus() {
        cout << "Manager Bonus: " << salary * 0.20 << endl;
    }

    void performanceReport() {
        cout << "Manager " << name << " has excellent leadership skills.\n";
    }

    void manageProject() {
        cout << "Manager is managing multiple projects.\n";
    }
};

class Developer : public Employee {
public:
    void calculateBonus() {
        cout << "Developer Bonus: " << salary * 0.15 << endl;
    }

    void performanceReport() {
        cout << "Developer " << name << " writes efficient code.\n";
    }

    void manageProject() {
        cout << "Developer is working on assigned modules.\n";
    }
};

class Programmer : public Employee {
public:
    void calculateBonus() {
        cout << "Programmer Bonus: " << salary * 0.10 << endl;
    }

    void performanceReport() {
        cout << "Programmer " << name << " is improving coding skills.\n";
    }

    void manageProject() {
        cout << "Programmer assists in development tasks.\n";
    }
};

int main() {
    Manager m;
    Developer d;
    Programmer p;

    m.setDetails("Rahul", "Delhi", 80000, "Manager");
    d.setDetails("Amit", "Mumbai", 60000, "Developer");
    p.setDetails("Neha", "Pune", 40000, "Programmer");

    cout << "\n--- Manager ---\n";
    m.calculateBonus();
    m.performanceReport();
    m.manageProject();

    cout << "\n--- Developer ---\n";
    d.calculateBonus();
    d.performanceReport();
    d.manageProject();

    cout << "\n--- Programmer ---\n";
    p.calculateBonus();
    p.performanceReport();
    p.manageProject();

    return 0;
}
