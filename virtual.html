{

class circle : public shape
{

    double rad;

    {
        rad = radius;
    }

    double calculateArea()
    {
        return M_PI * rad * rad;
    }

    {
        return (M_PI * 2 * rad);

{
private:
    double s1, s2, s3;

        s1 = side1;
        s2 = side2;
        s3 = side3;
    }

    double calculateArea()
    {
        double sum = ((s1 + s2 + s3) / 2);
        return (sqrt(sum * (sum - s1) * (sum - s2) * (sum - s3)));
    }
    double calculatePerimeter()

        return (s1 + s2 + s3);
    }
};

int main()
{

Q4:Write a C++ program to create an abstract class BankAccount with abstract methods deposit() and withdraw(). Create derived classes: SavingsAccount and CurrentAccount that extend the BankAccount class and implement the respective methods to handle deposits and withdrawals for each account type.


#include 
using namespace std;

class BankAccount
{
protected:
    int balance;

public:
    BankAccount() { balance = 0; }

    virtual void deposit() = 0;
    virtual void withdraw() = 0;

    void showBalance()
    {
        cout << "Current Balance: " << balance << endl;
    }
};

class SavingsAccount : public BankAccount
{
public:
    void deposit()
    {
        int amt;
        cout << "Enter amount to deposit: ";
        cin >> amt;
        balance += amt;
        cout << "Amount Deposited!\n";
        showBalance();
    }

    void withdraw()
    {
        int amt;
        cout << "Enter amount to withdraw: ";
        cin >> amt;

        // Minimum balance = 500
        if (balance - amt >= 500)
        {
            balance -= amt;
            cout << "Collect your cash\n";
        }
        else
        {
            cout << "Insufficient balance! Minimum balance should be 500\n";
        }
        showBalance();
    }
};

class CurrentAccount : public BankAccount
{
    int overdraftLimit;

public:
    CurrentAccount()
    {
        overdraftLimit = 1000; // allowed overdraft
    }

    void deposit()
    {
        int amt;
        cout << "Enter amount to deposit: ";
        cin >> amt;
        balance += amt;
        cout << "Amount Deposited!\n";
        showBalance();
    }

    void withdraw()
    {
        int amt;
        cout << "Enter amount to withdraw: ";
        cin >> amt;

        if (balance - amt >= -overdraftLimit)
        {
            balance -= amt;
            cout << "Collect your cash\n";
        }
        else
        {
            cout << "Overdraft limit exceeded!\n";
        }
        showBalance();
    }
};

int main()
{
    SavingsAccount sa;
    CurrentAccount ca;

    cout << "\n--- Savings Account ---\n";
    sa.deposit();
    sa.withdraw();

    cout << "\n--- Current Account ---\n";
    ca.deposit();
    ca.withdraw();

    return 0;
}


OUTPUT:
--- Savings Account ---
Enter amount to deposit: 2000
Amount Deposited!
Current Balance: 2000
Enter amount to withdraw: 200
Collect your cash
Current Balance: 1800

Q7:Write a C++ program to create an abstract class Shape3D with abstract methods calculateVolume() and calculateSurfaceArea(). Create derived classes Sphere and Cube that extend the Shape3D class and implement the respective methods to calculate the volume and surface area of each shape


#include 
using namespace std;

#define PI 3.14159

class Shape3D
{
public:
    virtual double calculateVolume() = 0;
    virtual double calculateSurfaceArea() = 0;
};

class Sphere : public Shape3D
{
private:
    double radius;

public:
    Sphere(double r)
    {
        radius = r;
    }

    double calculateVolume()
    {
        return (4.0 / 3) * PI * radius * radius * radius;
    }

    double calculateSurfaceArea()
    {
        return 4 * PI * radius * radius;
    }
};

class Cube : public Shape3D
{
private:
    double side;

public:
    Cube(double s)
    {
        side = s;
    }

    double calculateVolume()
    {
        return side * side * side;
    }

    double calculateSurfaceArea()
    {
        return 6 * side * side;
    }
};

int main()
{
    Shape3D *shape;

    Sphere s1(10);
    Cube c1(10);

    shape = &s1;
    cout << "Sphere Surface Area: " << shape->calculateSurfaceArea() << endl;
    cout << "Sphere Volume: " << shape->calculateVolume() << endl;

    cout << endl;

    shape = &c1;
    cout << "Cube Surface Area: " << shape->calculateSurfaceArea() << endl;
    cout << "Cube Volume: " << shape->calculateVolume() << endl;

    return 0;
}



OUTPUT:

Sphere Volume: 4188.79

        s1 = side1;

    }
    double perimeter()
    {
        return (s1 + s2 + s3);
    }
};
