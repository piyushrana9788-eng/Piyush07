Q5:Program to determine whether two matrices are equal 


#include
using namespace std;

int main()
{
    int row1,row2,column1,column2;

    cout<<"Enter rows and columns of matrix 1: ";
    cin>>row1>>column1;

    int matrix1[100][100];

    for(int i=0;i< row1;i++)
    {
        for(int j=0;j< column1;j++)
        {
            cin>>matrix1[i][j];
        }
    }

    cout<<"Enter rows and columns of matrix 2: ";
    cin>>row2>>column2;

    int matrix2[100][100];

    for(int i=0;i< row2;i++)
    {
        for(int j=0;j< column2;j++)
        {
            cin>>matrix2[i][j];
        }
    }

    if(row1==row2 && column1==column2)
    {
        bool same = true;  // ← small but critical change

        for(int i=0;i< row1;i++)
        {
            for(int j=0;j< column1;j++)
            {
                if(matrix1[i][j] != matrix2[i][j])
                {
                    same = false;
                    break;
                }
            }
            if(!same) break;
        }

        if(same)
            cout<<"Both matrices are same.";
        else
            cout<<"Matrices are not same.";
    }
    else
    {
        cout<<"Matrices are different in dimensions.";
    }

    return 0;
}

Q8:Program to find the frequency of odd & even numbers in the given matrix


#include
using namespace std;

int main()
{
    int rows, cols;

    cout<<"Enter number of rows and columns: ";
    cin>>rows>>cols;

    int matrix[100][100];

    cout<<"Enter elements:\n";
    for(int i=0;i< rows;i++)
    {
        for(int j=0;j< cols;j++)
        {
            cin>>matrix[i][j];
        }
    }

    int evenCount = 0, oddCount = 0;

    for(int i=0;i< rows;i++)
    {
        for(int j=0;j< cols;j++)
        {
            if(matrix[i][j] % 2 == 0)
                evenCount++;
            else
                oddCount++;
        }
    }

    cout<<"Even numbers: "<< evenCount<< endl;
    cout<<"Odd numbers: "<< oddCount<< endl;

    return 0;
}

Q10:Program to determine whether a given matrix is an identity matrix


#include
using namespace std;

int main()
{
    int n;

    cout<<"Enter size of square matrix: ";
    cin>>n;

    int matrix[100][100];

    cout<<"Enter elements:\n";
    for(int i=0;i< n;i++)
    {
        for(int j=0;j< n;j++)
        {
            cin>>matrix[i][j];
        }
    }

    bool isIdentity = true;

    for(int i=0;i< n;i++)
    {
        for(int j=0;j< n;j++)
        {
            if(i == j && matrix[i][j] != 1)
            {
                isIdentity = false;
                break;
            }
            else if(i != j && matrix[i][j] != 0)
            {
                isIdentity = false;
                break;
            }
        }
        if(!isIdentity) break;
    }

    if(isIdentity)
        cout<<"Matrix is an Identity Matrix";
    else
        cout<<"Matrix is NOT an Identity Matrix";

    return 0;
}

Q11:Program to Transpose matrix


#include 
using namespace std;

int main() {
    int rows, cols;

    cout << "Enter number of rows and columns: ";
    cin >> rows >> cols;

    int matrix[100][100], transpose[100][100];

    cout << "Enter matrix elements:\n";
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            cin >> matrix[i][j];
        }
    }

  
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            transpose[j][i] = matrix[i][j];
        }
    }

    cout << "\nOriginal Matrix:\n";
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }

    cout << "\nTranspose Matrix:\n";
    for(int i = 0; i < cols; i++) {
        for(int j = 0; j < rows; j++) {
            cout << transpose[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}

Q12:Program to determine whether a given matrix is a sparse matrix


#include 
using namespace std;

int main() {
    int rows, cols;
    
    cout << "Enter number of rows and columns: ";
    cin >> rows >> cols;

    int matrix[100][100];
    int zeroCount = 0, total;

    cout << "Enter matrix elements:\n";
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            cin >> matrix[i][j];
            if(matrix[i][j] == 0)
                zeroCount++;
        }
    }

    total = rows * cols;

    cout << "\nMatrix:\n";
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }

    if(zeroCount > total / 2)
        cout << "\nIt is a Sparse Matrix.\n";
    else
        cout << "\nIt is NOT a Sparse Matrix.\n";

    return 0;
}
